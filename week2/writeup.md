def extract_action_items_llm(text: str) -> List[str]:
    """
    Extract action items from text using Ollama's llama3 model, structured as a JSON array of strings.
    Requires the 'ollama' Python package and a running Ollama server.
    """
    import ollama
    import json

    system_prompt = (
        "Extract all action items from the following text. "
        "Respond ONLY with a JSON array of strings, where each string is an action item without extra explanation."
    )
    prompt = f"{text}\n\nReturn a JSON array of the action items only."
    
    response = ollama.chat(
        model="llama3",
        messages=[
            {"role": "system", "content": system_prompt},
            {"role": "user", "content": prompt}
        ],
        options={
            "response_format": "json"
        }
    )
    # Ollama's response should be a JSON array of strings as a string
    content = response.get("message", {}).get("content", "")
    try:
        items = json.loads(content)
        if isinstance(items, list) and all(isinstance(item, str) for item in items):
            return items
    except Exception:
        pass
    return []
